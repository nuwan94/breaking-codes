Capitalization Rule

Question

Given a word, you need to check whether the usage of capital letters in it is valid or not.

Major Capitalization Rules to Check
  • Rule #1 - All Upper
    Word is valid if all letters in word are capitals, like "UOK".

  • Rule #2 - All Lower
    Word is valid if all letters are non capitals, like "breakingcodes".

  • Rule #3 - First Letter Upper
    Otherwise word is valid only first letter in word is capital if there is more than one letter, like "Google".

Solution

  • C
  • Java 

word = input()

if len(word) > 1:
    print("VALID" if word.isupper() or word.islower()
          or word.istitle() else "NOT VALID")
else:
    print("VALID" if word.islower() else "NOT VALID")

                  

#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include <stdbool.h>

int main()
{

    char word[256];
    scanf("%s", word);

    bool isUpperCase = true;
    bool isLowerCase = true;
    bool isTitleCase = true;

    int len = strlen(word);
    int i;
    for (i = 0; i < len; i++)
    {
        char c = word[i];

        if (c < 'a' || c > 'z')
        {
            isLowerCase = false;
        }

        if (c < 'A' || c > 'Z')
        {
            isUpperCase = false;
        }

        if (i > 0 && (c > 'A' && c < 'Z'))
        {
            isTitleCase = false;
        }
    }

    if (len > 1)
    {
        printf((isUpperCase || isLowerCase || isTitleCase) ? "VALID" : "NOT VALID");
    }
    else
    {
        printf((isLowerCase) ? "VALID" : "NOT VALID");
    }

    return 0;
}

                    

import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);
        String word = input.next();

        if (word.length() > 1) {
            print((isUpper(word) || isLower(word) || isTitle(word)) ? "VALID" : "NOT VALID");
        } else {
            print(isLower(word) ? "VALID" : "NOT VALID");
        }
    }

    public static boolean isLower(String s) {
        return s == s.toLowerCase();
    }

    public static boolean isUpper(String s) {
        return s == s.toUpperCase();
    }

    public static boolean isTitle(String s) {
        char firstLetter = s.charAt(0);
        String restOfS = s.substring(1, s.length());
        return Character.isUpperCase(firstLetter) && isLower(restOfS);
    }

    public static void print(String msg) {
        System.out.print(msg);
    }
}
