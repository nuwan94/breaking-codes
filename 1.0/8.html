Fleet Battle

Question

Count the number of ships in given map. Map is represented using an \(n\) x \(m\) sized 2D Array. The ships are represented using '\(X\)'s, empty spaces are represented using '\(.\)'s.

Rules

  • You receive a valid map, only ships and empty spaces are in it.
  • Ships can only be placed horizontally or vertically.
  • There should be at least one horizontal or vertical space between two ships. So ships can't overlapped.

Sample Scenario 1
Input Format
3 4
X..X
...X
...X
Output Format
2

Solution

n, m = map(int, input().split()) board = []

for i in range(n):
    board.append(list(input()))

count = 0
for i in range(len(board)):
    j = 0
    while(j < len(board[0])):

        if (board[i][j] == 'X' and i == 0) or (board[i][j] == 'X' and i > 0 and board[i-1][j] == '.'):
            k = j+1
            while(k <= len(board[0]) - 1 and board[i][k] == 'X'):
                k += 1
            count += 1
            j = k
        else:
            j += 1

print(count)
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main(){
    int n, m;
    scanf("%d %d\n", &n, &m);

    char map[n][m];

    int i, j;
    for (i = 0; i < n; i++){
        for (j = 0; j < m; j++){
            char mapInp;
            scanf("%c ", &mapInp);
            map[i][j] = mapInp;
        }
    }

    int count = 0;

    for (i = 0; i < n; i++){
        j = 0;
        while (j < m){
            if ((map[i][j] == 'X' && i == 0) || (map[i][j] == 'X' && i > 0 && map[i - 1][j] == '.')){
                int k = j + 1;
                while (k <= m - 1 && map[i][k] == 'X'){
                    k += 1;
                }
                count += 1;
                j = k;
            }else{
                j += 1;
            }
        }
    }

    printf("%d", count);

    return 0;
} 
import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);
        int n = input.nextInt();
        int m = input.nextInt();

        char[][] map = new char[n][m];

        int i, j;
        for (i = 0; i < n; i++) {
            String row = input.next();
            for (j = 0; j < row.length(); j++) {
                map[i][j] = row.charAt(j);
            }
        }

        int count = 0;

        for (i = 0; i < n; i++) {
            j = 0;
            while (j < m) {
                if ((map[i][j] == 'X' && i == 0) || (map[i][j] == 'X' && i > 0 && map[i - 1][j] == '.')) {
                    int k = j + 1;
                    while (k <= m - 1 && map[i][k] == 'X') {
                        k += 1;
                    }
                    count += 1;
                    j = k;
                } else {
                    j += 1;
                }
            }
        }

        print(count);

    }

    public static void print(int msg) {
        System.out.print(msg);
    }
}
