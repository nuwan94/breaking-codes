Pascal's Triangle Generator

Question

Given a non-negative integer \(n\), generate the first \(n\) rows of Pascal's triangle.



Each number is the sum of the two numbers above it in Pascal's triangle.

Sample Scenario 1
Input Format
5
Output Format
    1
   1 1
  1 2 1
 1 3 3 1
1 4 6 4 1

Solution

n = int(input())

Pascal = []

for i in range(n):
    row = [1] * (i+1)
    Pascal.append(row)

for j in range(2, n):
    for k in range(1, j):
        Pascal[j][k] = Pascal[j-1][k-1] + Pascal[j-1][k]

for i in range(n):
    print(((n - i - 1) * " "), end='')
    print(*Pascal[i])
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main(){

    int a;
    scanf("%d", &a);
    int j = a;
    int k;
	
    for (int i = 0; i < a; i++){
	
        for (int p = a; p > i + 1; p--)
            printf(" ");
        
        k = 1;
        for (int j = 0; j <= i; j++){
            printf("%d ", k);
            k = k * (i - j) / (j + 1);
        }
        printf("\n");
    }
	
    return 0;
}
import java.io.*;
import java.util.*;

public class Solution {

    public static int pascal(int i, int j) {
        if ((j == 0) || (j == i))
            return 1;
        else
            return pascal(i - 1, j - 1) + pascal(i - 1, j);
    }

    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);
        int n = input.nextInt();

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n - i - 1; j++)
                print(" ");

            for (int j = 0; j <= i; j++)
                print(pascal(i, j) + ((j == i) ? "" : " "));

            print("\n");
        }

    }

    public static void print(String msg) {
        System.out.print(msg);
    }
}
