Sher Locked

Question

Sherlock and Watson are working on a new case, suddenly Sherlock got a encoded message from Professor Moriarty. After some investigations Sherlock figured that this encoded message can be decode by replacing the numbers with relevent alphabtic letters a=26, b=25, c=24.... y=02, z=01 likewise.

Help Sherlock by writing an algorithm to decode these messages.

Sample Scenario 1
Input Format
2
0822
250926161813202412232208
Output Format
se
brakingcodes

Solution

from textwrap import wrap

t = int(input())
for _ in range(t):
    inp = wrap(input(), 2)
    for i in inp:
        print(chr(ord('a') + (26 - int(i))), end='')
    print()
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>

int main(){

    int t = 0;
    scanf("%d", &t);

    int i;
    for (i = 0; i < t; i++){
        char word[256];
        scanf("%s", word);

        int p;
        for (p = 0; p < strlen(word); p += 2){
            char substr[3] = {word[p], word[p + 1]};
            printf("%c", 'a' + 26 - atoi(substr));
        }

        printf("\n");
    }

    return 0;
}

                    
import java.io.*;
import java.util.*;

public class Solution {

    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);
        int t = input.nextInt();

        for (int i = 0; i < t; i++) {
            String word = input.next();
            for (int p = 0; p < word.length(); p += 2) {
                int num = Integer.valueOf(word.substring(p, p + 2));
                print("" + (char) ('a' + 26 - num));
            }
            print("\n");
        }

    }

    public static void print(String msg) {
        System.out.print(msg);
    }
}

import java.io.*;

import java.util.regex.*;

public class Solution {

    public static void main(String[] args) {
        Scanner scan= new Scanner(System.in);
        int rounds= scan.nextInt();
        for(int i=0;i<rounds;i++){
            decode(scan.next());
        }
    }

    static void decode(String str){
        char []arr = str.toCharArray();
        
        for(int i=0;i<arr.length;i++){
            String number = Character.toString(arr[i])+Character.toString(arr[i+1]);
            int num = 123-Integer.parseInt(number);
            i++;
            System.out.print((char)num);
        }
        System.out.println();
    }

}
